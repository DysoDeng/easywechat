# 邀请成员

企业可通过接口批量邀请成员使用企业微信，邀请后将通过短信或邮件下发通知。

$params = [
    'user' => ['UserID1', 'UserID2', 'UserID3'],    // 成员ID列表, 最多支持1000个
    'party' => ['PartyID1', 'PartyID2'],            // 部门ID列表，最多支持100个
    'tag' => ['TagID1', 'TagID2'],                  // 标签ID列表，最多支持100个
];

$contacts->user->invite($params);
